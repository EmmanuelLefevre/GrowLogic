import { ChangeDetectionStrategy, Component } from '@angular/core';
import { RouterModule } from '@angular/router';
import { FontAwesomeModule } from '@fortawesome/angular-fontawesome';
import { TranslateModule } from '@ngx-translate/core';
import { MatTooltipModule } from '@angular/material/tooltip';

import { EXTERNAL_LINKS, FOOTER_SOCIAL_LINKS } from '@core/_config/links/social-links.constant';
import { HOSTING_INFOS } from '@core/_config/links/host-links.constant';

@Component({
  selector: 'main-footer',
  imports: [
    FontAwesomeModule,
    MatTooltipModule,
    RouterModule,
    TranslateModule
  ],
  templateUrl: './main-footer.component.html',
  styleUrl: './main-footer.component.scss',
  changeDetection: ChangeDetectionStrategy.OnPush
})

export class MainFooterComponent {
  protected readonly socialLinks = FOOTER_SOCIAL_LINKS;
  protected readonly externalLinks = EXTERNAL_LINKS;
  protected readonly hosting = HOSTING_INFOS;

  protected readonly currentYear = new Date().getFullYear();

  protected readonly appNameKey = 'META.DEFAULT.APP_NAME';
  protected readonly copyrightKey = 'LAYOUT.FOOTER.LEGAL.COPYRIGHT';
  protected readonly rightsReservedKey = 'LAYOUT.FOOTER.LEGAL.RIGHTS_RESERVED';
}

<footer class="footer">
  <ul class="footer__social-list">
    @for (link of socialLinks; track link.ariaKey) {
      <li class="footer__social-item">
        @if (link.path) {
          <a
            [routerLink]="[link.path]"
            [attr.aria-label]="link.ariaKey | translate"
            [matTooltip]="link.tooltipKey | translate"
            matTooltipPosition="above"
            matTooltipShowDelay="600"
            class="footer__link">
            <fa-icon
              [icon]="link.icon"
              class="footer__icon" />
          </a>
        } @else {
          <a
            [href]="link.href"
            target="_blank"
            rel="noopener noreferrer"
            [attr.aria-label]="link.ariaKey | translate"
            [matTooltip]="link.tooltipKey | translate"
            matTooltipPosition="above"
            matTooltipShowDelay="600"
            class="footer__link">
            <fa-icon
              [icon]="link.icon"
              class="footer__icon" />
          </a>
        }
      </li>
    }
  </ul>

  <p class="footer__copyright">
    {{ copyrightKey | translate: { year: currentYear } }}
    <span class="footer__owner">{{ appNameKey | translate }}</span>
    - {{ rightsReservedKey | translate }}
  </p>

  <div class="footer__hosting">
    <p class="footer__text">
      <span class="footer__group">
        <span class="footer__label">{{ hosting.developer.labelKey | translate }}</span>
        <span class="footer__value">
          <a
            [href]="externalLinks.portfolio"
            target="_blank"
            rel="noopener noreferrer"
            [attr.aria-label]="hosting.developer.ariaKey | translate"
            [matTooltip]="hosting.developer.tooltipKey | translate"
            matTooltipPosition="left"
            matTooltipShowDelay="600"
            >{{ hosting.developer.nameKey | translate }}
          </a>
        </span>
      </span>

      <span class="footer__separator"></span>

      <span class="footer__label">{{ hosting.host.labelKey | translate }}</span>

      <span class="footer__break"></span>

      <span class="footer__separator footer__separator--desktop"></span>

      <span class="footer__value">
        <a
          [href]="hosting.host.url"
          target="_blank"
          rel="noopener"
          [attr.aria-label]="hosting.host.ariaKey | translate"
          >{{ hosting.host.nameKey | translate }}
        </a>
      </span>

      <span class="footer__separator"></span>

      <span>{{ hosting.host.addressKey | translate }}</span>
    </p>

    <p class="footer__phone">
      <fa-icon
        [icon]="hosting.phone.icon"
        class="footer__phone_icon" />
      <span>{{ hosting.phone.nameKey | translate }}</span>
    </p>
  </div>
</footer>
